% ----------------------------------------
% Setup title design
% ----------------------------------------
\usepackage{titlesec}

% Remove number in header title 
\renewcommand{\sectionmark}[1]{\markboth{#1}{}}

% ----------------------------------------
% Chapter title design
% ----------------------------------------

\titleformat{\section}{\Large}{\thesection}{1em}{}
\titleformat{\subsection}{\large}{\thesubsection}{1em}{}
\titleformat{\subsubsection}{\normalfont}{\thesubsubsection}{1em}{}

% ----------------------------------------
% Table of contents design
% ----------------------------------------
\usepackage{tocloft}

% \makeatletter 
% \newcommand\cuslarge{\@setfontsize\cuslarge\@xxpt{13}}
% \makeatother

% Centering the table of contents title and make it uppercase
\renewcommand{\cfttoctitlefont}{\null\vspace{0.5cm}\hfill\fontsize{14.65pt}{14.65pt}\selectfont\MakeUppercase}

% Add page column title
\renewcommand{\cftaftertoctitle}{%
    % Note the second \hfill, it could have been any dummy text
    \hfill\hfill
    \\[1cm] \null\hfill
    \ifthenelse{\value{sprache} = 1}{%
        Seite%
    }{%
        Page%
    }
}
